# pickKeys

Reduces the wrapped object to only have the specified keys

```ts copy
import { obj } from "@tsly/obj";

const person = {
  first: "John",
  last: "Smith",
  age: 23,
  state: "NY",
};

const name = obj(person).dropKeys(["firstName", "lastName"]).take();

console.log(name); // { "firstName": "John", "lastName": "Smith" }
console.log(person); // { "firstName": "John", "lastName": "Smith", "age": 23, "state": "NY" }
```
